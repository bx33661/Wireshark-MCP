**Goal:** Add AI-powered anomaly detection (beacon detection, exfiltration indicators, protocol anomalies, baseline deviation) and a natural language query interface that maps security research intent to tshark operations.

**Tech Stack:** Python 3.10+, mcp (FastMCP), tshark, statistics stdlib, pytest + pytest-asyncio

```python
# src/wireshark_mcp/tools/anomaly.py
"""Anomaly detection tools for Wireshark MCP — statistical and heuristic analysis."""

import json
import logging
import statistics
from typing import Any

from ..tshark.client import TSharkClient
from .envelope import normalize_tool_result, parse_tool_result, success_response
from .formatting import CRIT, INFO, OK, WARN

logger = logging.getLogger("wireshark_mcp")


def _compute_jitter(intervals: list[float]) -> float:
    """Compute jitter coefficient (stddev / mean). Lower = more periodic."""
    if len(intervals) < 2:
        return 1.0
    mean = statistics.mean(intervals)
    if mean == 0:
        return 1.0
    return statistics.stdev(intervals) / mean


def _parse_tsv_rows(data: str) -> list[list[str]]:
    """Parse TSV output into rows of fields."""
    rows = []
    for line in data.strip().split("\n"):
        if line and not line.startswith("#"):
            rows.append(line.split("\t"))
    return rows


def make_contextual_anomaly_tools(client: TSharkClient) -> list[tuple[str, Any]]:
    """Create contextual anomaly detection tools."""

    async def wireshark_detect_beaconing(pcap_file: str, min_connections: int = 10, max_jitter: float = 0.2) -> str:
        """[Anomaly] Detect periodic communication patterns (C2 beacons) by analyzing connection timing intervals and jitter."""
        fields = ["ip.src", "ip.dst", "tcp.dstport", "frame.time_epoch"]
        result = await client.extract_fields(
            pcap_file,
            fields,
            display_filter="tcp.flags.syn == 1 && tcp.flags.ack == 0",
            limit=10000,
        )
        wrapped = parse_tool_result(result)
        if not wrapped["success"]:
            return normalize_tool_result(wrapped)

        raw_data = wrapped.get("data", "")
        rows = _parse_tsv_rows(raw_data)

        # Group connections by (src, dst, port)
        connections: dict[tuple[str, str, str], list[float]] = {}
        for row in rows:
            if len(row) >= 4:
                key = (row[0], row[1], row[2])
                try:
                    connections.setdefault(key, []).append(float(row[3]))
                except ValueError:
                    continue

        # Analyze each connection pair for periodicity
        beacons = []
        for (src, dst, port), timestamps in connections.items():
            if len(timestamps) < min_connections:
                continue
            timestamps.sort()
            intervals = [timestamps[i+1] - timestamps[i] for i in range(len(timestamps)-1)]
            if not intervals:
                continue
            jitter = _compute_jitter(intervals)
            if jitter <= max_jitter:
                mean_interval = statistics.mean(intervals)
                beacons.append({
                    "type": "beacon",
                    "severity": "high" if jitter < 0.05 else "medium",
                    "confidence": round(1.0 - jitter, 2),
                    "description": f"{src} -> {dst}:{port} every {mean_interval:.1f}s (jitter={jitter:.3f})",
                    "metadata": {
                        "src": src, "dst": dst, "port": port,
                        "interval_mean": round(mean_interval, 2),
                        "jitter": round(jitter, 4),
                        "session_count": len(timestamps),
                    },
                })

        output_parts = [f"{INFO} Beacon Detection Results"]
        if beacons:
            beacons.sort(key=lambda b: b["confidence"], reverse=True)
            output_parts.append(f"{CRIT} Found {len(beacons)} potential beacon(s):\n")
            for b in beacons[:20]:
                sev_icon = CRIT if b["severity"] == "high" else WARN
                output_parts.append(f"{sev_icon} [{b['confidence']}] {b['description']}")
            output_parts.append(f"\n{INFO} Structured findings:")
            output_parts.append(json.dumps(beacons[:20], indent=2))
        else:
            output_parts.append(f"{OK} No periodic beaconing patterns detected")

        return success_response("\n".join(output_parts))

    return [
        ("wireshark_detect_beaconing", wireshark_detect_beaconing),
    ]
```

```python
    async def wireshark_detect_exfiltration(pcap_file: str, limit: int = 10000) -> str:
        """[Anomaly] Detect potential data exfiltration (large outbound transfers, DNS query length anomalies, non-standard port usage)."""
        import asyncio

        # Check outbound TCP volume to external IPs
        tcp_task = client.extract_fields(
            pcap_file,
            ["ip.src", "ip.dst", "tcp.dstport", "tcp.len"],
            display_filter="tcp && !ip.dst == 10.0.0.0/8 && !ip.dst == 172.16.0.0/12 && !ip.dst == 192.168.0.0/16",
            limit=limit,
        )

        # Check for long DNS queries (potential DNS exfil)
        dns_task = client.extract_fields(
            pcap_file,
            ["ip.src", "dns.qry.name", "dns.qry.name.len"],
            display_filter="dns.qry.name.len > 50",
            limit=1000,
        )

        tcp_result, dns_result = await asyncio.gather(tcp_task, dns_task)

        output_parts = [f"{INFO} Data Exfiltration Analysis"]

        # Analyze TCP outbound volumes
        tcp_wrapped = parse_tool_result(tcp_result)
        if tcp_wrapped["success"] and tcp_wrapped.get("data"):
            rows = _parse_tsv_rows(tcp_wrapped["data"])
            # Aggregate bytes per (src, dst, port)
            volumes: dict[tuple[str, str, str], int] = {}
            for row in rows:
                if len(row) >= 4:
                    key = (row[0], row[1], row[2])
                    try:
                        volumes[key] = volumes.get(key, 0) + int(row[3])
                    except ValueError:
                        continue

            if volumes:
                sorted_vols = sorted(volumes.items(), key=lambda x: x[1], reverse=True)
                top_transfers = sorted_vols[:10]
                output_parts.append(f"\n{WARN} Top Outbound Transfers (by volume)")
                for (src, dst, port), total_bytes in top_transfers:
                    size_str = f"{total_bytes/1024:.1f} KB" if total_bytes < 1048576 else f"{total_bytes/1048576:.1f} MB"
                    output_parts.append(f"  {src} -> {dst}:{port} = {size_str}")

        # Analyze DNS query lengths
        dns_wrapped = parse_tool_result(dns_result)
        if dns_wrapped["success"] and dns_wrapped.get("data"):
            rows = _parse_tsv_rows(dns_wrapped["data"])
            if rows:
                output_parts.append(f"\n{CRIT} Long DNS Queries (potential DNS tunneling/exfil)")
                output_parts.append(f"  Found {len(rows)} queries with name length > 50 chars")
                for row in rows[:10]:
                    if len(row) >= 2:
                        output_parts.append(f"  {row[0]}: {row[1]}")

        return success_response("\n".join(output_parts))
```

```python
    async def wireshark_detect_protocol_anomalies(pcap_file: str, limit: int = 5000) -> str:
        """[Anomaly] Detect protocol anomalies (known protocols on non-standard ports, unusual protocol distributions)."""
        # Find known protocols on non-standard ports
        fields = ["ip.src", "ip.dst", "tcp.dstport", "_ws.col.Protocol"]
        result = await client.extract_fields(
            pcap_file,
            fields,
            display_filter="tcp && !tcp.dstport in {80 443 8080 8443}",
            limit=limit,
        )
        wrapped = parse_tool_result(result)
        if not wrapped["success"]:
            return normalize_tool_result(wrapped)

        rows = _parse_tsv_rows(wrapped.get("data", ""))

        # Find HTTP/TLS on non-standard ports
        suspicious_protocols = {"HTTP", "TLS", "SSL"}
        anomalies = []
        for row in rows:
            if len(row) >= 4 and row[3] in suspicious_protocols:
                anomalies.append({
                    "src": row[0], "dst": row[1],
                    "port": row[2], "protocol": row[3],
                })

        output_parts = [f"{INFO} Protocol Anomaly Detection"]
        if anomalies:
            # Deduplicate by (dst, port, protocol)
            seen = set()
            unique_anomalies = []
            for a in anomalies:
                key = (a["dst"], a["port"], a["protocol"])
                if key not in seen:
                    seen.add(key)
                    unique_anomalies.append(a)

            output_parts.append(f"\n{WARN} Known protocols on non-standard ports ({len(unique_anomalies)} unique)")
            for a in unique_anomalies[:20]:
                output_parts.append(f"  {a['src']} -> {a['dst']}:{a['port']} [{a['protocol']}]")
        else:
            output_parts.append(f"{OK} No protocol anomalies detected")

        return success_response("\n".join(output_parts))
```

```python
    async def wireshark_detect_anomalies(pcap_file: str, detectors: str = "all") -> str:
        """[Anomaly] Run all anomaly detectors concurrently (beacon, exfiltration, protocol). Returns combined findings."""
        import asyncio

        detector_map = {
            "beacon": wireshark_detect_beaconing,
            "exfiltration": wireshark_detect_exfiltration,
            "protocol": wireshark_detect_protocol_anomalies,
        }

        if detectors == "all":
            selected = list(detector_map.values())
        else:
            selected = [detector_map[d] for d in detectors.split(",") if d in detector_map]

        results = await asyncio.gather(*[d(pcap_file) for d in selected])

        output_parts = [f"{INFO} Comprehensive Anomaly Detection Report"]
        output_parts.append("=" * 50)
        for result in results:
            wrapped = parse_tool_result(result)
            if wrapped["success"]:
                output_parts.append(wrapped.get("data", ""))
                output_parts.append("-" * 50)

        return success_response("\n".join(output_parts))
```

```python
# src/wireshark_mcp/tools/nl_query.py
"""Natural language query engine for Wireshark MCP — maps intent to tshark operations."""

import json
import logging
from importlib import resources as importlib_resources
from pathlib import Path
from typing import Any

import yaml

from ..tshark.client import TSharkClient
from .envelope import success_response
from .formatting import INFO, WARN

logger = logging.getLogger("wireshark_mcp")

_TEMPLATES: list[dict[str, Any]] | None = None


def _load_templates() -> list[dict[str, Any]]:
    """Load intent templates from bundled + user custom YAML."""
    global _TEMPLATES
    if _TEMPLATES is not None:
        return _TEMPLATES

    templates: list[dict[str, Any]] = []

    # Load bundled templates
    try:
        data_ref = importlib_resources.files("wireshark_mcp") / "data" / "nl_templates.yaml"
        with importlib_resources.as_file(data_ref) as fp:
            data = yaml.safe_load(fp.read_text())
            templates.extend(data.get("templates", []))
    except Exception:
        logger.warning("Could not load bundled NL templates")

    # Load user custom templates
    user_file = Path.home() / ".wireshark-mcp" / "nl_templates.yaml"
    if user_file.exists():
        try:
            data = yaml.safe_load(user_file.read_text())
            templates.extend(data.get("templates", []))
        except Exception:
            logger.warning("Could not load user NL templates")

    _TEMPLATES = templates
    return _TEMPLATES


def _match_intent(query: str) -> dict[str, Any] | None:
    """Match a natural language query to an intent template."""
    templates = _load_templates()
    query_lower = query.lower()

    best_match = None
    best_score = 0

    for template in templates:
        score = 0
        all_keywords = template.get("keywords", []) + template.get("keywords_zh", [])
        for keyword in all_keywords:
            if keyword.lower() in query_lower:
                score += len(keyword)  # Longer matches score higher

        if score > best_score:
            best_score = score
            best_match = template

    return best_match if best_score > 0 else None


def make_contextual_nl_tools(client: TSharkClient) -> list[tuple[str, Any]]:
    """Create contextual NL query tools."""

    async def wireshark_nl_query(pcap_file: str, query: str) -> str:
        """[AI] Natural language query — describe what you're looking for and get matched to the right analysis tools."""
        matched = _match_intent(query)

        if matched is None:
            output_parts = [
                f"{WARN} Could not interpret query: '{query}'",
                "",
                f"{INFO} Try describing what you're looking for using keywords like:",
                "  - C2, beacon, command and control",
                "  - lateral movement, pivot, spread",
                "  - exfiltration, data leak, large transfer",
                "  - malware, trojan, backdoor",
                "  - suspicious DNS, tunnel, DGA",
                "  - credential, password, login",
                "  - investigate, full analysis",
                "",
                "Or use standard Wireshark display filter syntax directly.",
            ]
            return success_response("\n".join(output_parts))

        operations = matched.get("operations", [])
        tool_names = [op["tool"] for op in operations]

        output = {
            "interpreted_as": matched["description"],
            "intent": matched["intent"],
            "tools_to_invoke": tool_names,
            "suggested_workflow": f"Run these tools in sequence on '{pcap_file}': {', '.join(tool_names)}",
        }

        output_parts = [
            f"{INFO} Query interpreted as: {matched['description']}",
            f"",
            f"{INFO} Recommended tools to run:",
        ]
        for tool_name in tool_names:
            output_parts.append(f"  - {tool_name}")
        output_parts.append("")
        output_parts.append(f"{INFO} Structured response:")
        output_parts.append(json.dumps(output, indent=2))

        return success_response("\n".join(output_parts))

    return [
        ("wireshark_nl_query", wireshark_nl_query),
    ]
```
